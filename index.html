Gramatica

programa ::= declaraciones | sentencias
declaraciones ::= declaración | declaraciones | λ
declaración ::= tipo identificadores ;
tipo ::= entero | real
identificadores ::= identificador | identificador , identificadores
identificador ::= [a-zA-Z_][a-zA-Z0-9_]*
sentencias ::= sentencia | sentencias | λ
sentencia ::= asignación | operación | imprimir | leer
expresión ::= suma
suma ::= término ( ( "+" | "-" ) término )*
término ::= operando ( ( "*" | "/" ) operando )*
operando ::= identificador | número | "(" expresión ")"
número ::= [0-9]+(\.[0-9]+)?
operación ::= expresión operador expresión
operador ::= + | - | * | / | =
imprimir ::= cout << expresión ;
leer ::= cin >> identificador;
